3   import java.awt.Color;
4   import java.awt.Font;
5   import java.awt.Graphics2D;
6   import java.awt.event.MouseEvent;
7   import java.util.ArrayList;
8   import liisan_muistipeli.logic.Global;
9   
10  /**
11   * Luokka loppuvalikolle ja sen ilmoituksille.
12   */
13  public class EndMenu
14  {
15      private ArrayList<Button> buttons;
16      private Global global;
17      
18      //alignment variables
19      private int column0_from_left;
20      private int column1_from_left;
21      private int headline_fontsize;
22      private int fontsize;
23      private Font headline_font;
24      private Font font;
25      
26      public EndMenu(Global global)
27      {
28          column0_from_left = (int) (global.getVerticalsize()*0.2);
29          column1_from_left = (int) (global.getVerticalsize()*0.7);
30          headline_fontsize = 40;
31          fontsize = 24;
32          font = new Font("Monospaced", Font.PLAIN, fontsize);
33          headline_font = new Font("Monospaced", Font.PLAIN, headline_fontsize);
34          
35          this.global = global;
36          buttons = new ArrayList<>();
37          createToMainMenuButton();
38      }
39      
40      /**
41       * piirtää pelin tuloksen jframeen.
42       * @param g
43       * @param score 
44       */
45      public void drawScore(Graphics2D g, int score)
46      {
47          int fromleft = 100;
48          int fromtop = 100;
49          //int fontsize = 80;
50          
51          //Font font = new Font("Sherif", Font.PLAIN, fontsize);
52          g.setFont(headline_font);
53          g.setColor(Color.yellow);
54          g.drawString("score: "+score+"%", fromleft, fromtop);
55  
56      }
57      
58      /**
59       * luo palaa aloitusvalikkoon napin.
60       */
61      public void createToMainMenuButton()
62      {
63          int fromleft = column0_from_left;
64          int fromtop = (int) (global.getVerticalsize()*0.4);
65          
66          int width = (int) (global.getVerticalsize()*0.4);
67          int height = (int) (global.getVerticalsize()*0.1);
68          
69          //int fontsize = (int) (global.getHorizontalsize()*0.05);
70          
71          
72          buttons.add(new Button(fromleft, fromtop, width, height, fontsize, "main menu", 0, font));
73      }
74      
75      public ArrayList<Button> buttons()
76      {
77          return buttons;
78      }
79      
80      /**
81       * kokeilee onko hiiri jonkin napin kohdalla.
82       * @param me 
83       */
84      public void mousehover(MouseEvent me)
85      {
86          for (Button button : buttons)
87          {
88              button.mousehover(me);
89          }
90      }
91      
92      /**
93       * klikkaa nappia
94       * @param me
95       * @return button/null
96       */
97      public Button mouseclicked(MouseEvent me)
98      {
99          for (Button button : buttons)
100         {
101            if (button.mousehover(me)) return button;
102         }
103         return null;
104     }
105 }
