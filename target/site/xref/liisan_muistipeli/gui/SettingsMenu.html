13  public class SettingsMenu
14  {
15      private ArrayList<Button> buttons;

32      public SettingsMenu(Global global)
33      {
34          column0_from_left = (int) (global.getVerticalsize()*0.2);
35          column1_from_left = (int) (global.getVerticalsize()*0.8);
36          headline_fontsize = 40;
37          fontsize = 24;
38          font = new Font("Monospaced", Font.PLAIN, fontsize);
39          headline_font = new Font("Monospaced", Font.PLAIN, headline_fontsize);
40          
41          resolutions = new int[5][2];
42          
43          resolutions[0][0] = 1024;
44          resolutions[0][1] = 768;
45          resolutions[1][0] = 800;
46          resolutions[1][1] = 600;
47          resolutions[2][0] = 1280;
48          resolutions[2][1] = 720;
49          resolutions[3][0] = 1920;
50          resolutions[3][1] = 1080;
51          resolutions[4][0] = 1600;
52          resolutions[4][1] = 900;
53          
54          resolution_index = global.getResolution();
55          
56          difficulties = new String[3];
57          
58          difficulties[0] = "kid's stuff";
59          difficulties[1] = "normal";
60          difficulties[2] = "autistic";
61          
62          difficulties_index = global.getDifficulty();
63          
64          
65          xreso = global.getHorizontalsize();
66          yreso = global.getVerticalsize();
67          buttons = new ArrayList<>();
68          this.global = global;
69          createSetResolutionButton();
70          createReturnButton();
71          createApplyButton();
72          createDifficultyButton();
73      }
74      public int[] getResolution()
75      {
76          int[] value = new int[2];
77          value[0] = resolutions[resolution_index][0];
78          value[1] = resolutions[resolution_index][1];
79          return value;
80      }
81      public int getDifficulty()
82      {
83          return difficulties_index;
84      }

86      public ArrayList<Button> buttons()

90      public void drawHeadline(Graphics2D g)
91      {
92          //Font font = new Font("Monospaced", Font.PLAIN, 80);
93          g.setFont(headline_font);
94          g.setColor(Color.DARK_GRAY);
95          g.drawString("muistipeli/settings", 100, 100);
96      }
97      public void createDifficultyButton()
98      {
99          int fromleft = column0_from_left;
100         int fromtop = (int) (global.getVerticalsize()*0.6);
101         
102         int width = (int) (global.getVerticalsize()*0.25);
103         int height = (int) (global.getVerticalsize()*0.03);
104         
105         buttons.add(new Button(fromleft, fromtop, width, height, fontsize, "difficulty", 12, font));
106     }

140     public void showCurrentResolution(Graphics2D g)
141     {
142         String reso = resolutions[resolution_index][0]+" x "+resolutions[resolution_index][1];
143         g.setFont(font);
144         
145         int fromleft = (int) column1_from_left;
146         int fromtop = (int) (global.getVerticalsize()*0.65);
147 
148         g.drawString(reso, fromleft , fromtop);
149     }
150     public void showCurrentDifficulty(Graphics2D g)
151     {
152         String difficulty = difficulties[difficulties_index];
153         g.setFont(font);
154         
155         int fromleft = column1_from_left;
156         int fromtop = (int) (global.getVerticalsize()*0.6);
157     
158         g.drawString(difficulty, fromleft , fromtop);
159     }
160     public void changeDifficulty()
161     {
162         difficulties_index++;
163         if (difficulties_index > 2) difficulties_index = 0;
164         global.setDifficulty(difficulties_index);
165     }
166     
167     public void changeResolution()
168     {
169         resolution_index += 1;
170         if (resolution_index > 4) resolution_index = 0;
171         global.setResolution(resolution_index);
172     }
173     public void mousehover(MouseEvent me)
174     {
175         for (Button button : buttons)
176         {
177             button.mousehover(me);
178         }
179     }
180     public Button mouseclicked(MouseEvent me)
181     {
182         for (Button button : buttons)
183         {
184            if (button.mousehover(me)) return button;
185         }
186         return null;
187     }
